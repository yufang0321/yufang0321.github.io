TypeScript 私有類別屬性練習

兩個參數 A、B，A如果為"1"則B=true，A=“0"則B=false 其他則為null A可以任意輸入，但是B只能有限輸出

import { Injectable } from '@angular/core';

@Injectable({
  providedIn: 'root',
})
export class test {
  public a: string;
  private b: boolean;

  constructor(x: string) {
    this.a = x;
  }
  getB() {
    console.log('b:', this.b);
  }

  changeBByA() {
    console.log('a:', this.a);
    if (this.a === '1') {
      this.b = true;
    } else if (this.a === '0') {
      this.b = false;
    }
  }
}


import { Component, inject, ViewEncapsulation } from '@angular/core';
import { test } from './test/test';
@Component({
  selector: 'first-test',
  standalone: true,
  templateUrl: './first-test.html',
  encapsulation: ViewEncapsulation.None,
})
export class firstTest {
  constructor() {}
  ngOnInit() {
    let test1 = new test('0');
    test1.changeBByA(); 
    test1.getB(); //b: false
    
    let test2 = new test('1');
    test2.changeBByA(); 
    test2.getB(); //b: true

    let test3 = new test('3');
    test3.changeBByA(); 
    test3.getB(); //b: undefined
  }
}
